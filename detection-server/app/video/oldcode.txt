import os
import subprocess
from fastapi import FastAPI, HTTPException
from fastapi.responses import FileResponse

app = FastAPI()

HLS_FOLDER = "/tmp/hls"
HLS_PLAYLIST = os.path.join(HLS_FOLDER, "stream.m3u8")
ffmpeg_process = None

FFMPEG_CMD_TEMPLATE = [
    "ffmpeg",
    "-rtsp_transport", "tcp",
    "-i", "",  # Placeholder for RTSP URL
    "-c:v", "copy",
    "-c:a", "aac",
    "-f", "hls",
    "-hls_time", "2",
    "-hls_list_size", "5",
    "-hls_flags", "delete_segments",
    "-hls_segment_filename", os.path.join(HLS_FOLDER, "segment_%03d.ts"),
    HLS_PLAYLIST
]

@app.post("/start_stream/")
async def start_stream(rtsp_url: str):
    global ffmpeg_process

    if ffmpeg_process and ffmpeg_process.poll() is None:
        raise HTTPException(status_code=400, detail="Stream already running")

    if not os.path.exists(HLS_FOLDER):
        os.makedirs(HLS_FOLDER)

    # Clear previous HLS files
    for file in os.listdir(HLS_FOLDER):
        os.remove(os.path.join(HLS_FOLDER, file))

    cmd = FFMPEG_CMD_TEMPLATE.copy()
    cmd[4] = rtsp_url

    ffmpeg_process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)

    return {"status": "started"}

@app.post("/stop_stream/")
async def stop_stream():
    global ffmpeg_process
    if ffmpeg_process:
        ffmpeg_process.terminate()
        ffmpeg_process = None
        return {"status": "stopped"}
    else:
        raise HTTPException(status_code=400, detail="No active stream")

@app.get("/hls/{filename}")
async def hls_file(filename: str):
    filepath = os.path.join(HLS_FOLDER, filename)
    if not os.path.isfile(filepath):
        raise HTTPException(404, detail="File not found")
    media_type = "application/vnd.apple.mpegurl" if filename.endswith(".m3u8") else "video/MP2T"
    return FileResponse(filepath, media_type=media_type)

@app.get("/playlist")
async def get_playlist():
    if not os.path.exists(HLS_PLAYLIST):
        raise HTTPException(404, detail="Playlist not found")
    return FileResponse(HLS_PLAYLIST, media_type="application/vnd.apple.mpegurl")
